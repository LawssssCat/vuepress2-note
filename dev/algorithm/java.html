import java.util.Scanner;

class Stack {
    int[] data;
    int i = -1;
    public Stack(int size) {
        data = new int[size];
    }
    public void push(int v) {
        data[++i] = v;
    }
    public int pop() {
        return data[i--];
    }
    public int size() {
        return i + 1;
    }
}

            a: while((c = (char)in.read()) != '\n') {

                        break;
                    default: temp2 = temp2 * 10 + c - '0'; break;
                }
            }
            jisuan1(TEMPCHAR);

我的（常规实现：双栈）【不是最简洁的，有非常大优化空间】

import java.util.Scanner;

class Stack {
    Object[] data;;
    int index =-1;
    public Stack(int size) {
        data = new Object[size];
    }
    public void reverse() {
        for(int i=0; i<size()/2; i++) {
            Object t = data[i];
            int e = size()-1-i;
            data[i] = data[e];
            data[e] = t;
        }
    }
    public void clean() {
        index = -1;
    }
    public int size() {
        return index+1;
    }
    public boolean isEmpty() {
        return size()==0;
    }

    public void push(Object v) {
        index ++;
        data[index] = v;
    }
    public Object pop() {
        index --;
        return data[index+1];
    }
    public Object peek() {
        return data[index];
    }
}
